IN: /dat/nwp/wxmap/prc/wxmap:

1)	create area.AAAA.cfg   -- AAAA is the new name

see wxmap.doc.htm#webarea for
details on the format of the .cfg file

2)	create basemap:

	t.wxmap.pl basemap AAA

3)	create the button:

	wxmap.button.area.pl AAA title1 title2

4)	edit wxmap.env.pl and modify


$WXMAP{'WXMAP_AREAS'}       ="asia:bigaus:tropwpac:tropnio:tropsio:tropswpac:tropepac";
$WXMAP{'WXMAP_AREAS_TYPES'} ="midlat:special:tropic:tropic:tropic:tropic:tropic";
$WXMAP{'WXMAP_AREAS_DESC'}  ="East Asia:Big Aus:Tropical WPAC:Tropical NIO:Tropical SIO:Tropical SWPAC:Tropical EPAC";

    there is a one-to-one correspondence between each entry,
    e.g., asia->midlat->"East Asia"

    for control of plotting by area:

$plot_control{'ngp','tropsio','plots'} = $plots_nhem_tropical_season_full_wav;
$plot_control{'ngp','tropsio','taus'} = "default";

$plot_control{'avn','tropsio','plots'} = $plots_nhem_tropical_season_full;
$plot_control{'avn','tropsio','taus'} = "default";

$plot_control{'mrf','tropsio','plots'} = $plots_nhem_tropical_season_full;
$plot_control{'mrf','tropsio','taus'} = "default";

    if tc info add,
				      
$url{'tc','tropsio'}="http://www.solar.ifa.hawaii.edu/Tropical/tropical.html";
